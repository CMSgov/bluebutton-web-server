#!/usr/bin/env python 

# compatible with Python2.7 - 3.5 

from argparse import ArgumentParser 

from django.utils.crypto import pbkdf2 

import binascii 

import scrypt 

 

__author__ = "Alan Viars" 

 

 

def hcin_hash(hcin, salt, iterations=26000): 

return binascii.hexlify(pbkdf2(hcin, salt, iterations)).decode("ascii") 

 

 

def scrypt_hash(hcin, salt, n=2**20, r=8, p=1): 

return binascii.hexlify(scrypt.hash(hcin, salt, N=n, r=8, p=1)).decode("ascii") 

 

if __name__ == "__main__": 

parser = ArgumentParser("Perform a pbkdf2 one-way hash with a salt") 

parser.add_argument("hcin", help="The value to be hashed") 

parser.add_argument("salt", help="The salt to be applied to the hash") 

parser.add_argument("-a", "--algorithm", dest="algorithm", 

help="Algorithm, is either pbkdf2 or scrypt", 

default="pbkdf2") 

parser.add_argument("-i", "--iterations", dest="iterations", 

metavar='N', type=int, 

help="Number of iterations for the pbkdf2 hash", 

default=26000) 

parser.add_argument("-n", dest="n", 

metavar='n', type=int, 

help="n for Scrypt", 

default=2**20) 

parser.add_argument("-r", dest="r", 

metavar='r', type=int, 

help="r for Scrypt", 

default=8) 

parser.add_argument("-p", dest="p", 

metavar='p', type=int, 

help="p for Scrypt", 

default=1) 

args = parser.parse_args() 

if args.algorithm == 'pbkdf2': 

print(hcin_hash(args.hcin, args.salt, args.iterations)) 

elif args.algorithm == 'scrypt': 

print(scrypt_hash(args.hcin, args.salt, args.n, args.r, args.p))
